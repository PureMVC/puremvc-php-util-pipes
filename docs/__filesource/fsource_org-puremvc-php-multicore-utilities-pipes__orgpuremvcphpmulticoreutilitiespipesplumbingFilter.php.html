  27. /**
  28.  * Pipe Filter.
  29.  * <P>
  30.  * Filters may modify the contents of messages before writing them to
  31.  * their output pipe fitting. They may also have their parameters and
  32.  * filter function passed to them by control message, as well as having
  33.  * their Bypass/Filter operation mode toggled via control message.</p>
  34.  * 
  35.  * @package org.puremvc.php.multicore.utilities.pipes
  36.  */ 
  37. class Filter extends Pipe
  38. {
  39.     /**
  40.      * @var string 
  41.      */
  42.     protected $mode = FilterControlMessage::FILTER;
  43.  
  44.     /**
  45.      * @var function 
  46.      */
  47.     protected $filter = null; //default to function($message, $params){return;};
  48.     
  49.     /**
  50.      * @var object 
  51.      */
  52.     protected $params = null;
  53.     
  54.     /**
  55.      * @var string 
  56.      */
  57.     protected $name;
  58.     
  59.     /**
  60.      * Constructor.
  61.      * <P>
  62.      * Optionally connect the output and set the parameters.</P>
  63.      * 
  64.      * @param string $name 
  65.      * @param IPipeFitting $output 
  66.      * @param function $filter 
  67.      * @param object $params 
  68.      * @return Filter 
  69.      */
  70.     public function Filter( $name, IPipeFitting $output=null, $filter=null, $params=null ) 
  71.     {
  72.         parent::__construct( $output );
  73.         $this->name = $name;
  74.         $this->setFilter( $filter );
  75.         $this->setParams( $params );
  76.     }
  77.  
  78.     /**
  79.      * Handle the incoming message.
  80.      * <P>
  81.      * If message type is normal, filter the message (unless in BYPASS mode)
  82.      * and write the result to the output pipe fitting if the filter
  83.      * operation is successful.</P>
  84.      * 
  85.      * <P>
  86.      * The FilterControlMessage.SET_PARAMS message type tells the Filter
  87.      * that the message class is FilterControlMessage, which it
  88.      * casts the message to in order to retrieve the filter parameters
  89.      * object if the message is addressed to this filter.</P>
  90.      * 
  91.      * <P>
  92.      * The FilterControlMessage.SET_FILTER message type tells the Filter
  93.      * that the message class is FilterControlMessage, which it
  94.      * casts the message to in order to retrieve the filter function.</P>
  95.      * 
  96.      * <P>
  97.      * The FilterControlMessage.BYPASS message type tells the Filter
  98.      * that it should go into Bypass mode operation, passing all normal
  99.      * messages through unfiltered.</P>
  100.      * 
  101.      * <P>
  102.      * The FilterControlMessage.FILTER message type tells the Filter
  103.      * that it should go into Filtering mode operation, filtering all
  104.      * normal normal messages before writing out. This is the default
  105.      * mode of operation and so this message type need only be sent to
  106.      * cancel a previous BYPASS message.</P>
  107.      * 
  108.      * <P>
  109.      * The Filter only acts on the control message if it is targeted
  110.      * to this named filter instance. Otherwise it writes through to the
  111.      * output.</P>
  112.      * 
  113.      * @param IPipeMessage $message 
  114.      * @return Boolean True if the filter process does not throw an error and subsequent operations
  115.      *  in the pipeline succede.
  116.      */
  117.     public function write( IPipeMessage $message )
  118.     {
  119.         $outputMessage = null;
  120.         $success = true;
  121.  
  122.         // Filter normal messages
  123.         switch ( $message->getType())
  124.         {
  125.             case  Message::NORMAL:     
  126.                 try 
  127.                 {
  128.                     if ( $this->mode == FilterControlMessage::FILTER ) 
  129.                     {
  130.                         $outputMessage = $this->applyFilter( $message );
  131.                     } 
  132.                     else 
  133.                     {
  134.                         $outputMessage = $message;
  135.                     }
  136.                     $success = $this->output->write( $outputMessage );
  137.                 } 
  138.                 catch (Exception $e) 
  139.                 {
  140.                     $success = false;
  141.                 }
  142.                 break;
  143.             
  144.             // Accept parameters from control message 
  145.             case FilterControlMessage::SET_PARAMS:
  146.                 if ($this->isTarget($message))                     
  147.                 {
  148.                     $this->setParams( $message->getParams() );
  149.                 } 
  150.                 else 
  151.                 {
  152.                     $success = $this->output->write( $outputMessage );
  153.                 }
  154.                 break;
  155.  
  156.             // Accept filter function from control message 
  157.             case FilterControlMessage::SET_FILTER:
  158.                 if ($this->isTarget($message))
  159.                 {
  160.                     $this->setFilter( $message->getFilter() );
  161.                 } 
  162.                 else 
  163.                 {
  164.                     $success = $this->output->write( $outputMessage );
  165.                 }
  166.                 
  167.                 break;
  168.  
  169.             // Toggle between Filter or Bypass operational modes
  170.             case FilterControlMessage::BYPASS:
  171.             case FilterControlMessage::FILTER:
  172.                 if ($this->isTarget($message))
  173.                 {
  174.                     $this->mode = $message->getType();
  175.                 } 
  176.                 else 
  177.                 {
  178.                     $success = $this->output->write( $outputMessage );
  179.                 }
  180.                 break;
  181.             
  182.             // Write control messages for other fittings through
  183.             default:
  184.                 $success = $this->output->write( $outputMessage );
  185.         }
  186.         return $success;            
  187.     }
  188.     
  189.     /**
  190.      * Is the message directed at this filter instance?
  191.      * 
  192.      * @param IPipeMessage $message 
  193.      * @return bool 
  194.      */
  195.     protected function isTarget($message)
  196.     {
  197.         return ( $message->getName() == $this->name );
  198.     }
  199.     
  200.     /**
  201.      * Set the Filter parameters.
  202.      * <P>
  203.      * This can be an object can contain whatever arbitrary
  204.      * properties and values your filter method requires to
  205.      * operate.</P>
  206.      * 
  207.      * @param object $params the parameters object
  208.      */
  209.     public function setParams( $params )
  210.     {
  211.         $this->params = $params;
  212.     }
  213.  
  214.     /**
  215.      * Set the Filter function.
  216.      * <P>
  217.      * It must accept two arguments; an IPipeMessage,
  218.      * and a parameter Object, which can contain whatever
  219.      * arbitrary properties and values your filter method
  220.      * requires.</P>
  221.      * 
  222.      * @param function $filter the filter function.
  223.      * @return void 
  224.      */
  225.     public function setFilter( $filter )
  226.     {
  227.         $this->filter = (is_null($filter) ? (version_compare(phpversion(), '5.3.0', '<') ? create_function('$message, $params','return;') : function($message, $params){return;}) : $filter);
  228.     }
  229.     
  230.     /**
  231.      * Filter the message.
  232.      * 
  233.      * @param IPipeMessage $message 
  234.      * @return IPipeMessage 
  235.      */
  236.     protected function applyFilter( IPipeMessage $message )
  237.     {
  238.         $filter = $this->filter;
  239.         
  240.         $filter($message, $this->params);
  241.         return $message;
  242.     }
  243.     
  244. }
