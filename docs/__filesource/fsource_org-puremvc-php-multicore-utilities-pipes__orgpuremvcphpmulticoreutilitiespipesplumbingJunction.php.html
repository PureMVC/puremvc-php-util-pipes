  24. /**
  25.  * Pipe Junction.
  26.  * 
  27.  * <P>
  28.  * Manages Pipes for a Module.
  29.  * 
  30.  * <P>
  31.  * When you register a Pipe with a Junction, it is
  32.  * declared as being an INPUT pipe or an OUTPUT pipe.</P>
  33.  * 
  34.  * <P>
  35.  * You can retrieve or remove a registered Pipe by name,
  36.  * check to see if a Pipe with a given name exists,or if
  37.  * it exists AND is an INPUT or an OUTPUT Pipe.</P>
  38.  * 
  39.  * <P>
  40.  * You can send an <code>IPipeMessage</code> on a named INPUT Pipe
  41.  * or add a <code>PipeListener</code> to registered INPUT Pipe.</P>
  42.  * 
  43.  * @package org.puremvc.php.multicore.utilities.pipes
  44.  */
  45. class Junction
  46. {
  47.     /**
  48.      * INPUT Pipe Type
  49.      */
  50.     const INPUT = 'input';
  51.  
  52.     /**
  53.      * OUTPUT Pipe Type
  54.      */
  55.     const OUTPUT = 'output';
  56.     
  57.     /**
  58.      * The names of the INPUT pipes
  59.      * @var array 
  60.      */
  61.     protected $inputPipes = array();
  62.     
  63.     /**
  64.      * The names of the OUTPUT pipes
  65.      * @var array 
  66.      */
  67.     protected $outputPipes = array();
  68.     
  69.     /** 
  70.      * The map of pipe names to their pipes
  71.      * @var array 
  72.      */
  73.     protected $pipesMap = array();
  74.     
  75.     /**
  76.      * The map of pipe names to their types
  77.      * @var array 
  78.      */
  79.     protected $pipeTypesMap = array();
  80.  
  81.     // Constructor. 
  82.     public function __construct( )
  83.     {
  84.     }
  85.  
  86.     /**
  87.      * Register a pipe with the junction.
  88.      * <P>
  89.      * Pipes are registered by unique name and type,
  90.      * which must be either <code>Junction.INPUT</code>
  91.      * or <code>Junction.OUTPUT</code>.</P>
  92.       * <P>
  93.      * NOTE: You cannot have an INPUT pipe and an OUTPUT
  94.      * pipe registered with the same name. All pipe names
  95.      * must be unique regardless of type.</P>
  96.      * 
  97.      * @param string $name 
  98.      * @param string $type 
  99.      * @param IPipeFitting $pipe 
  100.      * @return bool true if successfully registered. false if another pipe exists by that name.
  101.      */
  102.     public function registerPipe( $name, $type, IPipeFitting $pipe )
  103.     { 
  104.         $success = true;
  105.         if ( !isset($this->pipesMap[$name]) )
  106.         {
  107.             $this->pipesMap[$name] = $pipe;
  108.             $this->pipeTypesMap[$name] = $type;
  109.             switch ($type) {
  110.                 case Junction::INPUT:
  111.                     array_push($this->inputPipes,$name);    
  112.                     break;                        
  113.                 case Junction::OUTPUT:
  114.                     array_push($this->outputPipes,$name);    
  115.                     break;                    
  116.                 default:    
  117.                     $success = false;
  118.             }
  119.         } 
  120.         else 
  121.         {
  122.             $success = false;
  123.         }
  124.         
  125.         return $success;
  126.     }
  127.     
  128.     /**
  129.      * Does this junction have a pipe by this name?
  130.      * 
  131.      * @param string $name the pipe to check for
  132.      * @return Boolean whether as pipe is registered with that name.
  133.      */ 
  134.     public function hasPipe( $name )
  135.     {
  136.         return isset($this->pipesMap[ $name ]);
  137.     }
  138.     
  139.     /**
  140.      * Does this junction have an INPUT pipe by this name?
  141.      * 
  142.      * @param string $name the pipe to check for
  143.      * @return bool Whether an INPUT pipe is registered with that name.
  144.      */ 
  145.     public function hasInputPipe( $name )
  146.     {
  147.         return ( $this->hasPipe($name) && ($this->pipeTypesMap[$name] == Junction::INPUT) );
  148.     }
  149.  
  150.     /**
  151.      * Does this junction have an OUTPUT pipe by this name?
  152.      * 
  153.      * @param string $name the pipe to check for
  154.      * @return bool Whether an OUTPUT pipe is registered with that name.
  155.      */ 
  156.     public function hasOutputPipe( $name )
  157.     {
  158.         return ( $this->hasPipe($name) && ($this->pipeTypesMap[$name] == Junction::OUTPUT) );
  159.     }
  160.  
  161.     /**
  162.      * Remove the pipe with this name if it is registered.
  163.      * <P>
  164.      * NOTE: You cannot have an INPUT pipe and an OUTPUT
  165.      * pipe registered with the same name. All pipe names
  166.      * must be unique regardless of type.</P>
  167.      * 
  168.      * @param string $name the pipe to remove
  169.      * @retuirn void
  170.      */
  171.     public function removePipe( $name ) 
  172.     {
  173.         if ( $this->hasPipe($name) ) 
  174.         {
  175.              $type = $this->pipeTypesMap[$name];
  176.              $pipesList = array();
  177.             switch ($type) {
  178.                 case Junction::INPUT:
  179.                     $pipesList = $this->inputPipes;
  180.                     break;                        
  181.                 case Junction::OUTPUT:
  182.                     $pipesList = $this->outputPipes;    
  183.                     break;                    
  184.             }
  185.             for ( $i=0; $i<count($pipesList); $i++)
  186.             {
  187.                 if ($pipesList[$i] == $name)
  188.                 {
  189.                     array_slice($pipesList,$i,1);
  190.                     break;
  191.                 }
  192.             }
  193.             unset($this->pipesMap[$name]);
  194.             unset($this->pipeTypesMap[$name]);
  195.         }
  196.     }
  197.  
  198.     /**
  199.      * Retrieve the named pipe.
  200.      * 
  201.      * @param string $name the pipe to retrieve
  202.      * @return IPipeFitting the pipe registered by the given name if it exists
  203.      */
  204.     public function retrievePipe( $name ) 
  205.     {
  206.         return (isset($this->pipesMap[$name]) ? $this->pipesMap[$name] : null);
  207.     }
  208.  
  209.     /**
  210.      * Add a PipeListener to an INPUT pipe.
  211.      * <P>
  212.      * NOTE: there can only be one PipeListener per pipe,
  213.      * and the listener function must accept an IPipeMessage
  214.      * as its sole argument.</P>
  215.      * 
  216.      * @param string $inputPipeName the INPUT pipe to add a PipeListener to
  217.      * @param object $context the calling context or 'this' object
  218.      * @param string $listener the function name on the context to call
  219.      * @return bool 
  220.      */
  221.     public function addPipeListener( $inputPipeName, $context, $listener ) 
  222.     {
  223.         $success = false;
  224.         if ( $this->hasInputPipe($inputPipeName) )
  225.         {
  226.             $pipe = $this->pipesMap[$inputPipeName];
  227.             $success = $pipe->connect( new PipeListener($context, $listener) );
  228.         } 
  229.         return $success;
  230.     }
  231.     
  232.     /**
  233.      * Send a message on an OUTPUT pipe.
  234.      * 
  235.      * @param string $outputPipeName the OUTPUT pipe to send the message on
  236.      * @param IPipeMessage $message the IPipeMessage to send
  237.      * @return bool 
  238.      */
  239.     public function sendMessage( $outputPipeName, IPipeMessage $message ) 
  240.     {
  241.         $success = false;
  242.         if ( $this->hasOutputPipe($outputPipeName) )
  243.         {
  244.             $pipe = $this->pipesMap[$outputPipeName];
  245.             $success = $pipe->write($message);
  246.         } 
  247.         return $success;
  248.     }
  249.  
  250. }
