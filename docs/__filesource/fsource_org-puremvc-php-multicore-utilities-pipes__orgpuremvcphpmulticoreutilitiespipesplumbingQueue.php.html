  26.  
  27. /** 
  28.  * Pipe Queue.
  29.  * <P>
  30.  * The Queue always stores inbound messages until you send it
  31.  * a FLUSH control message, at which point it writes its buffer
  32.  * to the output pipe fitting. The Queue can be sent a SORT
  33.  * control message to go into sort-by-priority mode or a FIFO
  34.  * control message to cancel sort mode and return the
  35.  * default mode of operation, FIFO.</P>
  36.  * 
  37.  * <P>
  38.  * NOTE: There can effectively be only one Queue on a given
  39.  * pipeline, since the first Queue acts on any queue control
  40.  * message. Multiple queues in one pipeline are of dubious
  41.  * use, and so having to name them would make their operation
  42.  * more complex than need be.</P>
  43.  * 
  44.  * @package org.puremvc.php.multicore.utilities.pipes
  45.  */
  46. class Queue extends Pipe
  47. {
  48.     /**
  49.      * @var string 
  50.      */
  51.     protected $mode = QueueControlMessage::FIFO;
  52.     
  53.     /**
  54.      * 
  55.      * @var array 
  56.      */
  57.     protected $messages = array();
  58.     
  59.     public function Queue( IPipeFitting $output=null )
  60.     {
  61.         parent::__construct( $output );
  62.     }
  63.     
  64.     /**
  65.      * Handle the incoming message.
  66.      * <P>
  67.      * Normal messages are enqueued.</P>
  68.      * <P>
  69.      * The FLUSH message type tells the Queue to write all
  70.      * stored messages to the ouptut PipeFitting, then
  71.      * return to normal enqueing operation.</P>
  72.      * <P>
  73.      * The SORT message type tells the Queue to sort all
  74.      * <I>subsequent</I> incoming messages by priority. If there
  75.      * are unflushed messages in the queue, they will not be
  76.      * sorted unless a new message is sent before the next FLUSH.
  77.      * Sorting-by-priority behavior continues even after a FLUSH,
  78.      * and can be turned off by sending a FIFO message, which is
  79.      * the default behavior for enqueue/dequeue.</P>
  80.      *  
  81.       * @param IPipeMessage $message 
  82.       * @return bool 
  83.       */
  84.     public function write( IPipeMessage $message )
  85.     {
  86.         $success = true;
  87.         switch ( $message->getType() )    
  88.         {
  89.             // Store normal messages
  90.             case Message::NORMAL:
  91.                 $this->store( $message );
  92.                 break;
  93.                 
  94.             // Flush the queue
  95.             case QueueControlMessage::FLUSH:
  96.                 $success = $this->flush();        
  97.                 break;
  98.  
  99.             // Put Queue into Priority Sort or FIFO mode 
  100.             // Subsequent messages written to the queue
  101.             // will be affected. Sorted messages cannot
  102.             // be put back into FIFO order!
  103.             case QueueControlMessage::SORT:
  104.             case QueueControlMessage::FIFO:
  105.                 $this->mode = $message->getType();
  106.                 break;
  107.         }
  108.         return $success;
  109.     } 
  110.     
  111.     /**
  112.      * Store a message.
  113.      * @param message the IPipeMessage to enqueue.
  114.      * @return int the new count of messages in the queue
  115.      */
  116.     protected function store( IPipeMessage $message ) 
  117.     {
  118.         array_push($this->messages, $message );
  119.         
  120.         if ($this->mode == QueueControlMessage::SORT )
  121.         {
  122.             usort($this->messages,array($this, "sortMessagesByPriority"));
  123.         }
  124.     }
  125.  
  126.  
  127.     /**
  128.      * Sort the Messages by priority.
  129.      */
  130.     protected function sortMessagesByPriority(IPipeMessage $msgA, IPipeMessage $msgB)
  131.     {
  132.         if ($msgA->getPriority() == $msgB->getPriority()) 
  133.         {
  134.             return 0;
  135.         }
  136.         return ($msgA->getPriority() < $msgB->getPriority()) ? -1 : 1;
  137.     }
  138.     
  139.     /**
  140.      * Flush the queue.
  141.      * <P>
  142.      * NOTE: This empties the queue.</P>
  143.      * @return bool Boolean true if all messages written successfully.
  144.      */
  145.     protected function flush()
  146.     {
  147.         $success = true;
  148.         $message = array_shift($this->messages);
  149.         while ( $message != null ) 
  150.         {
  151.             $ok = $this->output->write( $message );
  152.             if ( !$ok )
  153.             {
  154.                 $success = false;
  155.             }
  156.             $message = array_shift($this->messages);
  157.         } 
  158.         return $success;
  159.     }
  160.  
  161. }
